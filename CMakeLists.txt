cmake_minimum_required(VERSION 3.15)
project(KeyMemRT)

# This CMakeLists.txt is used to discover system dependencies and generate
# a build.config file for the Makefile. It does NOT build the project itself.
# The Makefile handles the actual build process.

message(STATUS "KeyMemRT dependency discovery")
message(STATUS "========================================")

# Find OpenFHE - check common install locations
if(EXISTS "/usr/include/openfhe")
    set(OPENFHE_INCLUDES "-I/usr/include -I/usr/include/openfhe -I/usr/include/openfhe/pke -I/usr/include/openfhe/core -I/usr/include/openfhe/binfhe")
    set(OPENFHE_LIBS "-lOPENFHEcore -lOPENFHEpke -lOPENFHEbinfhe")
    message(STATUS "✓ Found OpenFHE at /usr/include/openfhe")
elseif(EXISTS "/usr/local/include/openfhe")
    set(OPENFHE_INCLUDES "-I/usr/local/include -I/usr/local/include/openfhe -I/usr/local/include/openfhe/pke -I/usr/local/include/openfhe/core -I/usr/local/include/openfhe/binfhe")
    set(OPENFHE_LIBS "-L/usr/local/lib -lOPENFHEcore -lOPENFHEpke -lOPENFHEbinfhe")
    message(STATUS "✓ Found OpenFHE at /usr/local/include/openfhe")
else()
    set(OPENFHE_INCLUDES "")
    set(OPENFHE_LIBS "")
    message(WARNING "✗ OpenFHE not found")
endif()

# Find cereal (header-only library)
# The KeyMemRT compiler generates code with paths like "include/cereal/..."
# so we need to add -I/usr to resolve this
find_path(CEREAL_INCLUDE_DIR
    NAMES cereal/cereal.hpp
    PATHS /usr/include /usr/include/openfhe /usr/local/include /usr/local/include/openfhe
    DOC "Path to cereal headers"
)

if(CEREAL_INCLUDE_DIR)
    message(STATUS "✓ Found cereal: ${CEREAL_INCLUDE_DIR}")
    set(CEREAL_INCLUDES "-I${CEREAL_INCLUDE_DIR}")

    # IMPORTANT: Generated code uses "include/cereal/..." paths
    # Add /usr so that "include/cereal/..." resolves to "/usr/include/cereal/..."
    set(CEREAL_INCLUDES "${CEREAL_INCLUDES} -I/usr")
else()
    message(STATUS "✗ cereal not found, using fallback paths")
    set(CEREAL_INCLUDES "-I/usr/include -I/usr/local/include/cereal -I/usr")
endif()

# Generate Makefile configuration
set(CONFIG_FILE "${CMAKE_CURRENT_BINARY_DIR}/build.config")
file(WRITE ${CONFIG_FILE} "# Auto-generated by CMake - DO NOT EDIT\n")
file(APPEND ${CONFIG_FILE} "# Run 'cmake -B build' to regenerate\n\n")
file(APPEND ${CONFIG_FILE} "# OpenFHE configuration\n")
file(APPEND ${CONFIG_FILE} "CMAKE_OPENFHE_INCLUDES := ${OPENFHE_INCLUDES}\n")
file(APPEND ${CONFIG_FILE} "CMAKE_OPENFHE_LIBS := ${OPENFHE_LIBS}\n\n")
file(APPEND ${CONFIG_FILE} "# Cereal configuration\n")
file(APPEND ${CONFIG_FILE} "CMAKE_CEREAL_INCLUDES := ${CEREAL_INCLUDES}\n\n")
file(APPEND ${CONFIG_FILE} "# Indicate CMake configuration is available\n")
file(APPEND ${CONFIG_FILE} "CMAKE_CONFIGURED := 1\n")

message(STATUS "========================================")
message(STATUS "✓ Generated ${CONFIG_FILE}")
message(STATUS "  Makefile will use these discovered paths")
message(STATUS "========================================")
